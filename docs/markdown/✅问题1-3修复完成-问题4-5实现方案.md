```php
<?php

namespace App\Services;

use App\Models\Employee;
use App\Models\OnboardingForm;
use Dompdf\Dompdf;
use Dompdf\Options;

class OnboardingFormPdfService
{
    public function generatePdf($employeeId)
    {
        $employee = Employee::with('onboardingForm')->findOrFail($employeeId);
        $form = $employee->onboardingForm;
        
        if (!$form) {
            throw new \Exception('该员工未填写入职登记表');
        }
        
        // 准备PDF内容
        $html = view('pdf.onboarding_form', [
            'employee' => $employee,
            'form' => $form
        ])->render();
        
        // 配置Dompdf
        $options = new Options();
        $options->set('isHtml5ParserEnabled', true);
        $options->set('isRemoteEnabled', true);
        $options->set('defaultFont', 'simhei'); // 中文字体
        
        $dompdf = new Dompdf($options);
        $dompdf->loadHtml($html);
        $dompdf->setPaper('A4', 'portrait');
        $dompdf->render();
        
        return $dompdf->output();
    }
    
    public function generateMultiplePdfs(array $employeeIds)
    {
        $zip = new \ZipArchive();
        $zipFileName = 'employee_onboarding_forms_' . date('YmdHis') . '.zip';
        $zipPath = storage_path('app/temp/' . $zipFileName);
        
        if (!is_dir(storage_path('app/temp'))) {
            mkdir(storage_path('app/temp'), 0755, true);
        }
        
        if ($zip->open($zipPath, \ZipArchive::CREATE) !== true) {
            throw new \Exception('无法创建ZIP文件');
        }
        
        foreach ($employeeIds as $employeeId) {
            try {
                $employee = Employee::findOrFail($employeeId);
                $pdfContent = $this->generatePdf($employeeId);
                
                $fileName = $employee->name . '_入职登记表.pdf';
                $zip->addFromString($fileName, $pdfContent);
            } catch (\Exception $e) {
                \Log::error("生成PDF失败: {$employeeId}", ['error' => $e->getMessage()]);
                continue;
            }
        }
        
        $zip->close();
        
        return $zipPath;
    }
}
```

#### 4. 添加控制器方法 `EmployeeController.php`

```php
use App\Services\OnboardingFormPdfService;

/**
 * 批量导出入职登记表PDF
 */
public function exportOnboardingFormsPdf(Request $request)
{
    $validator = Validator::make($request->all(), [
        'employee_ids' => 'required|array',
        'employee_ids.*' => 'integer|exists:employees,id',
    ]);

    if ($validator->fails()) {
        return response()->json([
            'success' => false,
            'message' => '参数错误',
            'errors' => $validator->errors()
        ], 422);
    }

    try {
        $pdfService = new OnboardingFormPdfService();
        
        if (count($request->employee_ids) === 1) {
            // 单个员工直接下载PDF
            $employeeId = $request->employee_ids[0];
            $employee = Employee::findOrFail($employeeId);
            $pdfContent = $pdfService->generatePdf($employeeId);
            
            $fileName = $employee->name . '_入职登记表.pdf';
            
            return response($pdfContent, 200, [
                'Content-Type' => 'application/pdf',
                'Content-Disposition' => 'attachment; filename="' . $fileName . '"',
            ]);
        } else {
            // 多个员工打包成ZIP
            $zipPath = $pdfService->generateMultiplePdfs($request->employee_ids);
            
            return response()->download($zipPath)->deleteFileAfterSend(true);
        }
    } catch (\Exception $e) {
        \Log::error('导出PDF失败: ' . $e->getMessage());
        return response()->json([
            'success' => false,
            'message' => '导出失败: ' . $e->getMessage()
        ], 500);
    }
}
```

```vue
<template>
  <!-- 在操作按钮区域添加 -->
  <el-button

    @click="handleBatchExportPdf"
    :disabled="!selectedEmployees.length"
  >

  </el-button>

    :data="employees"

<script setup>
import { ref } from 'vue'
import { ElMessage } from 'element-plus'
import request from '@/api/request'

const selectedEmployees = ref([])

const handleSelectionChange = (selection) => {
  selectedEmployees.value = selection
}

const handleBatchExportPdf = async () => {
  if (!selectedEmployees.value.length) {
    ElMessage.warning('请先选择员工')
    return
  }
  
  try {
    ElMessage.info('正在生成PDF，请稍候...')
    
    const response = await request({
      url: '/employees/export-onboarding-pdfs',
      method: 'post',
      data: {
        employee_ids: selectedEmployees.value.map(e => e.id)
      },
      responseType: 'blob'
    })
    
    // 创建下载链接
    const url = window.URL.createObjectURL(new Blob([response]))
    const link = document.createElement('a')
    link.href = url
    
    // 设置文件名
    const fileName = selectedEmployees.value.length === 1
      ? `${selectedEmployees.value[0].name}_入职登记表.pdf`
      : `入职登记表_${new Date().getTime()}.zip`
    
    link.setAttribute('download', fileName)
    document.body.appendChild(link)
    link.click()
    document.body.removeChild(link)
    window.URL.revokeObjectURL(url)
    
    ElMessage.success('导出成功')
  } catch (error) {
    console.error('导出PDF失败:', error)
    ElMessage.error('导出失败')
  }
}
</script>
```
