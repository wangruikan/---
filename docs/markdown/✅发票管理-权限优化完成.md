#### JavaScript 权限检查
```javascript
import request from '@/api/request'
import { useAccountSetStore } from '@/stores/accountSet'

// 账套store
const accountSetStore = useAccountSetStore()

// 权限控制：是否可以创建任务（只有审批人可以）
const canCreateTask = ref(false)

// 检查创建权限（只有审批人可以创建任务）
const checkCreatePermission = async () => {
  try {
    const accountSetId = accountSetStore.currentAccountSetId
    if (!accountSetId) {
      canCreateTask.value = false
      return
    }
    
    const response = await request({
      url: '/check-delivery-config-access',
      method: 'get',
      params: { account_set_id: accountSetId }
    })
    
    if (response.success) {
      // 只有审批级别2、3、4可以创建任务
      canCreateTask.value = response.has_access
    } else {
      canCreateTask.value = false
    }
  } catch (error) {
    console.error('检查创建权限失败:', error)
    canCreateTask.value = false
  }
}

// 初始化
onMounted(() => {
  loadData()
  loadInvoiceProjects()
  checkCreatePermission() // ← 新增权限检查
})
```
