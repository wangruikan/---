## PC端实现（Vue3 + Element Plus）

### 1. 在人员档案中添加离职证明标签页

在 `src/views/Employees/index.vue` 的员工详情对话框中添加：

```vue
<template>
  <!-- 在员工详情对话框的 el-tabs 中添加 -->
  <el-tabs v-model="activeTab">
    <el-tab-pane label="员工信息" name="employee">
      <!-- 现有的员工信息表单 -->
    </el-tab-pane>
    
    <!-- 新增：离职证明标签页 -->
    <el-tab-pane 
      label="离职证明" 
      name="resignation"
      v-if="needsResignationCertificate"
    >
      <div class="resignation-certificate-section">
        <!-- 上传区域 -->
        <div class="upload-section" style="margin-bottom: 20px;">
          <el-upload
            :action="resignationUploadUrl"
            :headers="uploadHeaders"
            :on-success="handleResignationUploadSuccess"
            :on-error="handleResignationUploadError"
            :before-upload="beforeResignationUpload"
            :show-file-list="false"
            multiple
          >
            <el-button type="primary" :icon="Upload">
              上传离职证明
            </el-button>
          </el-upload>
          <div class="upload-tip">
            支持上传多个文件，单个文件最大10MB
          </div>
        </div>

        <!-- 文件列表 -->
        <el-table 
          :data="resignationCertificates" 
          v-loading="loadingCertificates"
          border
        >
          <el-table-column prop="file_name" label="文件名" min-width="200" />
          <el-table-column label="文件大小" width="120">
            <template #default="{ row }">
              {{ formatFileSize(row.file_size) }}
            </template>
          </el-table-column>
          <el-table-column label="上传来源" width="100">
            <template #default="{ row }">
              <el-tag :type="row.upload_source === 'miniprogram' ? 'success' : 'primary'" size="small">
                {{ row.upload_source === 'miniprogram' ? '小程序' : 'PC端' }}
              </el-tag>
            </template>
          </el-table-column>
          <el-table-column label="上传人" width="120">
            <template #default="{ row }">
              {{ row.uploader?.name || '-' }}
            </template>
          </el-table-column>
          <el-table-column label="上传时间" width="180">
            <template #default="{ row }">
              {{ formatDateTime(row.created_at) }}
            </template>
          </el-table-column>
          <el-table-column label="操作" width="150" fixed="right">
            <template #default="{ row }">
              <el-button 
                type="primary" 
                size="small" 
                @click="handleDownloadCertificate(row)"
              >
                下载
              </el-button>
              <el-button 
                type="danger" 
                size="small" 
                @click="handleDeleteCertificate(row)"
              >
                删除
              </el-button>
            </template>
          </el-table-column>
        </el-table>

        <!-- 空状态 -->
        <el-empty 
          v-if="!loadingCertificates && resignationCertificates.length === 0"
          description="暂无离职证明"
        />
      </div>
    </el-tab-pane>
  </el-tabs>
</template>

<script setup>
import { ref, computed } from 'vue'
import { ElMessage, ElMessageBox } from 'element-plus'
import { Upload } from '@element-plus/icons-vue'
import request from '@/api/request'

// 离职证明相关
const resignationCertificates = ref([])
const loadingCertificates = ref(false)

// 判断是否需要显示离职证明标签页
const needsResignationCertificate = computed(() => {
  if (!currentEmployee.value) return false
  const status = currentEmployee.value.contract_status
  return ['terminated', 'retired'].includes(status)
})

// 上传URL
const resignationUploadUrl = computed(() => {
  if (!currentEmployee.value) return ''
  const baseURL = import.meta.env.VITE_API_BASE_URL || ''
  return `${baseURL}/api/employees/${currentEmployee.value.id}/resignation-certificates/upload`
})

// 上传请求头
const uploadHeaders = computed(() => {
  const token = localStorage.getItem('token')
  return {
    'Authorization': `Bearer ${token}`
  }
})

// 加载离职证明列表
const loadResignationCertificates = async (employeeId) => {
  if (!employeeId) return
  
  try {
    loadingCertificates.value = true
    const response = await request({
      url: `/employees/${employeeId}/resignation-certificates`,
      method: 'get'
    })
    resignationCertificates.value = response.data || []
  } catch (error) {
    console.error('加载离职证明失败:', error)
    ElMessage.error('加载离职证明失败')
  } finally {
    loadingCertificates.value = false
  }
}

// 上传前验证
const beforeResignationUpload = (file) => {
  const isLt10M = file.size / 1024 / 1024 < 10
  if (!isLt10M) {
    ElMessage.error('文件大小不能超过 10MB!')
    return false
  }
  return true
}

// 上传成功
const handleResignationUploadSuccess = (response) => {
  if (response.success) {
    ElMessage.success('上传成功')
    loadResignationCertificates(currentEmployee.value.id)
  } else {
    ElMessage.error(response.message || '上传失败')
  }
}

// 上传失败
const handleResignationUploadError = (error) => {
  console.error('上传失败:', error)
  ElMessage.error('上传失败')
}

// 下载离职证明
const handleDownloadCertificate = async (certificate) => {
  try {
    const baseURL = import.meta.env.VITE_API_BASE_URL || ''
    const url = `${baseURL}/api/employees/resignation-certificates/${certificate.id}/download`
    window.open(url, '_blank')
  } catch (error) {
    console.error('下载失败:', error)
    ElMessage.error('下载失败')
  }
}

// 删除离职证明
const handleDeleteCertificate = async (certificate) => {
  try {
    await ElMessageBox.confirm(
      `确定要删除 ${certificate.file_name} 吗？`,
      '确认删除',
      {
        confirmButtonText: '确定',
        cancelButtonText: '取消',
        type: 'warning'
      }
    )
    
    await request({
      url: `/employees/resignation-certificates/${certificate.id}`,
      method: 'delete'
    })
    
    ElMessage.success('删除成功')
    loadResignationCertificates(currentEmployee.value.id)
    
  } catch (error) {
    if (error !== 'cancel') {
      console.error('删除失败:', error)
      ElMessage.error('删除失败')
    }
  }
}

// 格式化文件大小
const formatFileSize = (bytes) => {
  if (!bytes) return '-'
  if (bytes < 1024) return bytes + ' B'
  if (bytes < 1024 * 1024) return (bytes / 1024).toFixed(2) + ' KB'
  return (bytes / 1024 / 1024).toFixed(2) + ' MB'
}

// 格式化日期时间
const formatDateTime = (datetime) => {
  if (!datetime) return '-'
  return new Date(datetime).toLocaleString('zh-CN')
}

// 在查看员工详情时加载离职证明
const handleViewEmployee = async (employee) => {
  currentEmployee.value = employee
  // ... 其他逻辑
  
  // 如果是离职/退休员工，加载离职证明
  if (['terminated', 'retired'].includes(employee.contract_status)) {
    loadResignationCertificates(employee.id)
  }
}
</script>
```

```javascript
// pages/resignation-certificate/index.js
const app = getApp()
const API_URL = app.globalData.apiUrl

Page({
  data: {
    files: [],
    loading: false,
    employeeId: null
  },

  onLoad() {
    // 获取当前登录员工ID
    const userInfo = wx.getStorageSync('userInfo')
    if (userInfo && userInfo.employee_id) {
      this.setData({ employeeId: userInfo.employee_id })
      this.loadFiles()
    } else {
      wx.showToast({
        title: '请先登录',
        icon: 'none'
      })
    }
  },

  // 加载文件列表
  loadFiles() {
    this.setData({ loading: true })
    
    wx.request({
      url: `${API_URL}/employees/${this.data.employeeId}/resignation-certificates`,
      method: 'GET',
      header: {
        'Authorization': `Bearer ${wx.getStorageSync('token')}`
      },
      success: (res) => {
        if (res.data.success) {
          const files = res.data.data.map(file => ({
            ...file,
            file_size_text: this.formatFileSize(file.file_size),
            created_at: this.formatDateTime(file.created_at)
          }))
          this.setData({ files })
        }
      },
      fail: (error) => {
        console.error('加载失败:', error)
        wx.showToast({
          title: '加载失败',
          icon: 'none'
        })
      },
      complete: () => {
        this.setData({ loading: false })
      }
    })
  },

  // 选择文件
  chooseFile() {
    wx.chooseMessageFile({
      count: 9,
      type: 'file',
      success: (res) => {
        const files = res.tempFiles
        
        // 检查文件大小
        const oversizeFiles = files.filter(f => f.size > 10 * 1024 * 1024)
        if (oversizeFiles.length > 0) {
          wx.showToast({
            title: '文件大小不能超过10MB',
            icon: 'none'
          })
          return
        }
        
        // 上传文件
        this.uploadFiles(files)
      }
    })
  },

  // 上传文件
  uploadFiles(files) {
    wx.showLoading({ title: '上传中...' })
    
    const uploadPromises = files.map(file => {
      return new Promise((resolve, reject) => {
        wx.uploadFile({
          url: `${API_URL}/employees/${this.data.employeeId}/resignation-certificates/upload`,
          filePath: file.path,
          name: 'file',
          formData: {
            source: 'miniprogram'
          },
          header: {
            'Authorization': `Bearer ${wx.getStorageSync('token')}`
          },
          success: (res) => {
            const data = JSON.parse(res.data)
            if (data.success) {
              resolve()
            } else {
              reject(data.message)
            }
          },
          fail: reject
        })
      })
    })
    
    Promise.all(uploadPromises)
      .then(() => {
        wx.hideLoading()
        wx.showToast({
          title: '上传成功',
          icon: 'success'
        })
        this.loadFiles()
      })
      .catch((error) => {
        wx.hideLoading()
        wx.showToast({
          title: error || '上传失败',
          icon: 'none'
        })
      })
  },

  // 下载文件
  downloadFile(e) {
    const id = e.currentTarget.dataset.id
    const file = this.data.files.find(f => f.id === id)
    
    if (!file) return
    
    wx.showLoading({ title: '下载中...' })
    
    wx.downloadFile({
      url: `${API_URL}/employees/resignation-certificates/${id}/download`,
      header: {
        'Authorization': `Bearer ${wx.getStorageSync('token')}`
      },
      success: (res) => {
        if (res.statusCode === 200) {
          wx.hideLoading()
          wx.openDocument({
            filePath: res.tempFilePath,
            success: () => {
              console.log('打开文档成功')
            }
          })
        }
      },
      fail: (error) => {
        wx.hideLoading()
        wx.showToast({
          title: '下载失败',
          icon: 'none'
        })
      }
    })
  },

  // 删除文件
  deleteFile(e) {
    const id = e.currentTarget.dataset.id
    const file = this.data.files.find(f => f.id === id)
    
    if (!file) return
    
    wx.showModal({
      title: '确认删除',
      content: `确定要删除 ${file.file_name} 吗？`,
      success: (res) => {
        if (res.confirm) {
          wx.request({
            url: `${API_URL}/employees/resignation-certificates/${id}`,
            method: 'DELETE',
            header: {
              'Authorization': `Bearer ${wx.getStorageSync('token')}`
            },
            success: (res) => {
              if (res.data.success) {
                wx.showToast({
                  title: '删除成功',
                  icon: 'success'
                })
                this.loadFiles()
              }
            },
            fail: () => {
              wx.showToast({
                title: '删除失败',
                icon: 'none'
              })
            }
          })
        }
      }
    })
  },

  // 格式化文件大小
  formatFileSize(bytes) {
    if (!bytes) return '-'
    if (bytes < 1024) return bytes + ' B'
    if (bytes < 1024 * 1024) return (bytes / 1024).toFixed(2) + ' KB'
    return (bytes / 1024 / 1024).toFixed(2) + ' MB'
  },

  // 格式化日期时间
  formatDateTime(datetime) {
    if (!datetime) return '-'
    const date = new Date(datetime)
    return `${date.getFullYear()}-${String(date.getMonth() + 1).padStart(2, '0')}-${String(date.getDate()).padStart(2, '0')}`
  }
})
```

1. 创建工资表，包含离职员工
2. 不上传离职证明
3. 提交审批 → 应该显示错误
4. 上传离职证明
5. 再次提交审批 → 应该成功

## 注意事项

1. **权限验证**：确保员工只能上传自己的离职证明
2. **文件大小**：限制单个文件最大10MB
3. **文件类型**：建议限制为PDF、图片等常见格式
4. **错误处理**：完善的错误提示和异常处理
5. **加载状态**：显示加载动画，提升用户体验
