### 2. index.js - 页面逻辑

```javascript
const app = getApp()

Page({
  data: {
    fileList: [],
    loading: false,
    employeeId: null
  },

  onLoad() {
    // 获取当前登录员工ID
    const userInfo = wx.getStorageSync('userInfo')
    if (userInfo && userInfo.employee_id) {
      this.setData({
        employeeId: userInfo.employee_id
      })
      this.loadFileList()
    } else {
      wx.showToast({
        title: '请先登录',
        icon: 'none'
      })
      setTimeout(() => {
        wx.navigateBack()
      }, 1500)
    }
  },

  // 加载文件列表
  loadFileList() {
    this.setData({ loading: true })
    
    wx.request({
      url: `${app.globalData.apiUrl}/employees/${this.data.employeeId}/resignation-certificates`,
      method: 'GET',
      header: {
        'Authorization': `Bearer ${wx.getStorageSync('token')}`
      },
      success: (res) => {
        if (res.data.success) {
          const fileList = res.data.data.map(item => ({
            ...item,
            file_size_text: this.formatFileSize(item.file_size),
            created_at: this.formatTime(item.created_at)
          }))
          this.setData({ fileList })
        } else {
          wx.showToast({
            title: res.data.message || '加载失败',
            icon: 'none'
          })
        }
      },
      fail: () => {
        wx.showToast({
          title: '网络错误',
          icon: 'none'
        })
      },
      complete: () => {
        this.setData({ loading: false })
      }
    })
  },

  // 选择文件
  chooseFile() {
    wx.chooseMessageFile({
      count: 5,
      type: 'file',
      extension: ['jpg', 'png', 'pdf'],
      success: (res) => {
        const files = res.tempFiles
        
        // 检查文件大小
        const oversizeFiles = files.filter(file => file.size > 10 * 1024 * 1024)
        if (oversizeFiles.length > 0) {
          wx.showToast({
            title: '文件大小不能超过 10MB',
            icon: 'none'
          })
          return
        }
        
        // 上传文件
        this.uploadFiles(files)
      }
    })
  },

  // 上传文件
  uploadFiles(files) {
    wx.showLoading({
      title: '上传中...',
      mask: true
    })
    
    let uploadCount = 0
    const totalCount = files.length
    
    files.forEach(file => {
      wx.uploadFile({
        url: `${app.globalData.apiUrl}/employees/${this.data.employeeId}/resignation-certificates/upload`,
        filePath: file.path,
        name: 'files[]',
        header: {
          'Authorization': `Bearer ${wx.getStorageSync('token')}`
        },
        success: (res) => {
          const data = JSON.parse(res.data)
          if (!data.success) {
            wx.showToast({
              title: data.message || '上传失败',
              icon: 'none'
            })
          }
        },
        fail: () => {
          wx.showToast({
            title: '上传失败',
            icon: 'none'
          })
        },
        complete: () => {
          uploadCount++
          if (uploadCount === totalCount) {
            wx.hideLoading()
            wx.showToast({
              title: '上传完成',
              icon: 'success'
            })
            // 重新加载列表
            this.loadFileList()
          }
        }
      })
    })
  },

  // 预览文件
  previewFile(e) {
    const item = e.currentTarget.dataset.item
    
    if (item.file_type === 'application/pdf') {
      // PDF 文件，下载后打开
      wx.showLoading({
        title: '加载中...',
        mask: true
      })
      
      wx.downloadFile({
        url: `${app.globalData.apiUrl}/employees/resignation-certificates/${item.id}/download`,
        header: {
          'Authorization': `Bearer ${wx.getStorageSync('token')}`
        },
        success: (res) => {
          if (res.statusCode === 200) {
            wx.openDocument({
              filePath: res.tempFilePath,
              fileType: 'pdf',
              success: () => {
                console.log('打开文档成功')
              }
            })
          }
        },
        fail: () => {
          wx.showToast({
            title: '加载失败',
            icon: 'none'
          })
        },
        complete: () => {
          wx.hideLoading()
        }
      })
    } else {
      // 图片文件，使用预览
      const imageUrl = `${app.globalData.apiUrl}/storage/${item.file_path}`
      wx.previewImage({
        urls: [imageUrl],
        current: imageUrl
      })
    }
  },

  // 删除文件
  deleteFile(e) {
    const id = e.currentTarget.dataset.id
    
    wx.showModal({
      title: '提示',
      content: '确定要删除这个文件吗？',
      success: (res) => {
        if (res.confirm) {
          wx.showLoading({
            title: '删除中...',
            mask: true
          })
          
          wx.request({
            url: `${app.globalData.apiUrl}/employees/resignation-certificates/${id}`,
            method: 'DELETE',
            header: {
              'Authorization': `Bearer ${wx.getStorageSync('token')}`
            },
            success: (res) => {
              if (res.data.success) {
                wx.showToast({
                  title: '删除成功',
                  icon: 'success'
                })
                // 重新加载列表
                this.loadFileList()
              } else {
                wx.showToast({
                  title: res.data.message || '删除失败',
                  icon: 'none'
                })
              }
            },
            fail: () => {
              wx.showToast({
                title: '网络错误',
                icon: 'none'
              })
            },
            complete: () => {
              wx.hideLoading()
            }
          })
        }
      }
    })
  },

  // 格式化文件大小
  formatFileSize(bytes) {
    if (bytes === 0) return '0 B'
    const k = 1024
    const sizes = ['B', 'KB', 'MB', 'GB']
    const i = Math.floor(Math.log(bytes) / Math.log(k))
    return (bytes / Math.pow(k, i)).toFixed(2) + ' ' + sizes[i]
  },

  // 格式化时间
  formatTime(dateString) {
    if (!dateString) return ''
    const date = new Date(dateString)
    const year = date.getFullYear()
    const month = String(date.getMonth() + 1).padStart(2, '0')
    const day = String(date.getDate()).padStart(2, '0')
    const hour = String(date.getHours()).padStart(2, '0')
    const minute = String(date.getMinutes()).padStart(2, '0')
    return `${year}-${month}-${day} ${hour}:${minute}`
  },

  // 下拉刷新
  onPullDownRefresh() {
    this.loadFileList()
    setTimeout(() => {
      wx.stopPullDownRefresh()
    }, 1000)
  }
})
```

### pages/mine/index.js

```javascript
Page({
  data: {
    showResignationCertificate: false
  },

  onLoad() {
    this.checkResignationCertificateAccess()
  },

  // 检查是否显示离职证明入口
  checkResignationCertificateAccess() {
    const userInfo = wx.getStorageSync('userInfo')
    if (userInfo && userInfo.contract_status) {
      const needsCertificate = ['terminated', 'retired'].includes(userInfo.contract_status)
      this.setData({
        showResignationCertificate: needsCertificate
      })
    }
  },

  // 跳转到离职证明页面
  goToResignationCertificate() {
    wx.navigateTo({
      url: '/pages/mine/resignation-certificate/index'
    })
  }
})
```

## 📝 app.js 配置

```javascript
App({
  globalData: {
    apiUrl: 'https://your-domain.com/api', // 替换为实际的API地址
    userInfo: null
  },

  onLaunch() {
    // 获取用户信息
    this.getUserInfo()
  },

  getUserInfo() {
    const token = wx.getStorageSync('token')
    if (token) {
      wx.request({
        url: `${this.globalData.apiUrl}/user/info`,
        method: 'GET',
        header: {
          'Authorization': `Bearer ${token}`
        },
        success: (res) => {
          if (res.data.success) {
            this.globalData.userInfo = res.data.data
            wx.setStorageSync('userInfo', res.data.data)
          }
        }
      })
    }
  }
})
```

## 🔧 注意事项

1. **API 地址配置**
   - 在 `app.js` 中配置正确的 API 地址
   - 确保小程序服务器域名已配置

2. **权限验证**
   - 确保用户已登录
   - Token 存储在本地存储中
   - 每次请求都携带 Authorization 头

3. **文件类型限制**
   - 支持 JPG、PNG、PDF 格式
   - 单个文件最大 10MB
   - 可以一次选择多个文件

4. **错误处理**
   - 网络错误提示
   - 文件大小超限提示
   - 上传失败提示

5. **用户体验**
   - 上传时显示加载状态
   - 支持下拉刷新
   - 文件预览功能
